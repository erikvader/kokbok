\begin{recipe}[
  portion={\portion{4}},
  bakingtime={\SI{30}{\minute}},
  source={Någon av Victorias kokböcker}
  ]{Familjens Curryfavorit}
  \graph{big = \recipepicture{linscurry}, bigpicturewidth = 0.3\textwidth}

  \introduction{En snabblagad gryta med många smaker som är lätta att gilla.
  Lägg till mer vatten (\SI{2}{dl}) för att få en soppa.}

  \ingredients{%
    1 & gul lök\\
    3 & vitlöksklyftor\\
    2 & morötter\\
    \SI{15}{ml}&curry\\
    \SI{2}{dl}&röda linser\\
    2&burkar kokosmjölk (à \SI{400}{ml})\\
    2&grönsaksbuljongtärningar\\
    \SI{3}{dl}&vatten\\
    \sfrac{1}{2}&blomkålshuvud\\
    lite&olja\\
    nypa&salt och peppar\\
    \SI{15}{ml}&garam masala\\
  }

  \preparation{%
    \step Skala och hacka lök, vitlök och morötter. Fräst mjukt i olja tillsammans med
    curryn och linserna, rör om ordentligt under tiden så att smakerna kommer fram och
    inget bränns vid.
    \step Häll i kokosmjölk, buljongtärningar och vatten. Koka under lock ca 10 minuter.
    \step Bryt blomkålen i buketter, lägg dem i grytan och låt puttra ytterligare 5
minuter. Smaka av med salt och peppar.
    }
\end{recipe}
