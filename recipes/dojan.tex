\begin{recipe}[
  portion={\portion{4}},
  bakingtime={\SI{15}{\minute}}
  ]{Dojanröra}
  \graph{big = \recipepicture{dojan}}

  \introduction{En rätt från Dojan när han slängde ihop lite godtyckliga ingredienser han
    hade hemma en gång.}

  \ingredients{%
    \SI{5}{dl} & crème fraiche\\
    lagom & spaghetti\\
    1 burk & majs på \SI{160}{\gram}\\
    2$\times$\SI{140}{\gram} & tärnad bacon \unsure{eller 3?}\\
    lite&dragon\\
  }

  \preparation{%
    \step Koka spaghettin enligt anvisningarna på förpackningen.
    \step Stek bacon. Lägg till majset mot slutet.\shortstep
    \step Ha i crème fraiche och låt det bli till en tjock sås. Krydda med dragon.
  }

  \hint{Skopa gärna ur det smälta fettet från baconet för att minska på snuskigheten.}
\end{recipe}
