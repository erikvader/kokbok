\begin{recipe}[
  portion={\portion{5}\unsure{4?}},
  bakingtime={\SI{45}{\minute}}
  ]{Köttfärssås Taco}
  \graph{big = \recipepicture{köttfärssås}}
  \ingredients{%
    \SI{800}{g} & krossade tomater\\
    1 & vitlölksklyfta\\
    1 & gul lök\\
    \SI{1000}{g} & köttfärs \\
    mängd\unsure{?} & spaghetti\\
    lite\unsure{?} & tomatpuré\\
    1 & köttbuljongstärning\\
    nypa&salt\\
    lite\unsure{?}&tabasco\\
    lite\unsure{?}&chilipulver\\
    lite\unsure{?}&tacokrydda\\
  }

  \preparation{%
    \step Koka pastan enligt anvisningen på förpackningen.
    \step Hacka och stek löken i lagom mängd olja.\shortstep
    \step Bryn köttfärsen tillsammans med löken. Tillsätt sedan tomatpuré och vitlök.
    \step Häll ner tomaten och buljongen. Krydda och låt puttra en stund.
  }
\end{recipe}
