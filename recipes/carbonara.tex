\begin{recipe}[
  portion={\portion{4}},
  bakingtime={\SI{30}{\minute}},
  source={\url{https://www.ica.se/recept/kramig-carbonara-722780/}}
  ]{``Carbonara''}
  \graph{big = \recipepicture{carbonara}}

  \introduction{Tvättäkta carbonara som skulle få italienarna att vända sig i sina gravar!}

  \ingredients{%
    \SI{100}{g} & finriven pecorino eller parmesan\\
    \SI{750}{g} & spaghetti\\
    6 & ägg\\
    \SI{3}{dl}&vispgrädde\\
    3$\times$\SI{140}{\gram} & tärnad bacon\\
    nypa&salt och peppar\\
  }

  \preparation{%
    \step Koka spaghettin enligt anvisningarna på förpackningen och stek bacon.
    \step Vispa ihop ägg, grädde, parmesan, salt och peppar i en bunke.
    \step Blanda ihop allting till en sörja.
  }

  \hint{Det är svårt att blanda allting jämnt och fint i en kastrull, så det är enklare
    att servera de tre delarna för sig.}
\end{recipe}
