\begin{recipe}[
  portion={\portion{1}},
  bakingtime={\SI{45}{\minute}},
  source=\url{https://www.ica.se/recept/pannkakor-grundsmet-2083/}
  ]{Pannkakor}
  \graph{big = \recipepicture{pannkakor}}

  \introduction{En väldigt enkel och omättande rätt.}

  \ingredients{%
    \SI{2.5}{dl} & vetemjöl\\
    \SI{6}{dl} & mjölk\\
    3 & ägg\\
    lite & smör eller olja \\
    \optional\\
    lite & glass\\
    lite&sylt\\
    lite&grädde\\
  }

  \preparation{%
    \step Blanda ihop allt till en smet\shortstep
    \step Stek i en stekpanna. Den första pannkakan blir alltid den bästa!
    \step Servera tillsammans med vad du vill
  }
\end{recipe}
