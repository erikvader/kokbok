\begin{recipe}[
  portion={\portion{4} \unsure{6 kanske?}},
  bakingtime={\SI{30}{\minute}},
  source={\portionenundertian},
  ]{Tandoorigryta}

  \graph{big = \recipepicture{tandoorigryta}}

  \ingredients{%
    \SI{4}{port} & matvete \unsure{dl?}\\
    1 & vitlöksklyfta \\
    \SI{2}{msk} & tandooripasta eller krydda\\
    \SI{1}{tsk} & sambal oelek\\
    \SI{2.5}{dl} & röda linser\\
    \SI{3}{dl} & vatten\\
    \SI{4}{dl} & kokosmjölk\\
    1 & grönsaksbuljongtärning\\
    2 & morötter\\
    2 & paprikabiter, grillad inlagd \unsure{eller?}\\
    lite & salt och peppar
  }

  \preparation{%
    \step Koka matvetet enligt anvisningen på förpackningen
    \step Hacka vitlök och fräs i olja tillsammans med sambal oelek och tandooripastan
    \step Lägg i linserna, vatten, kokosmjölk och buljong. Koka upp.
    \step Skär morötterna i kuber och paprikan i strimlor. Lägg i morötterna efter \SI{5}{\minute}. Lägg i paprikan efter ytterligare \SI{5}{\minute}.
    \step Koka i ytterligare \SI{5}{\minute} tills linserna är mjuka
  }
\end{recipe}
